    $(document).ready(function () {
        // Trigger in language dropdown for adding new language
        langSelector.change(function () {
            if (langSelector.val() && lvlSelector.val() > 0) {
                AddLanguage();
                if (submitted) {
                    $("form").valid();
                }
            }
        });

        // Trigger in language dropdown for adding new language
        lvlSelector.change(function () {
            if (lvlSelector.val() && lvlSelector.val() > 0) {
                AddLanguage();
                if (submitted) {
                    $("form").valid();
                }
            }
        });

        // Add preselected languages
        var items = JSON.parse(addedLanguagesSelector.val());
        $(".list_items").empty();
        for (var i = 0; i < items.length; i++) {
            $(".list_items").append(GenerateLanguageElement(items[i].Language, items[i].Level));
        }
        $('.icon.close').off("click");
        $('.icon.close').on("click", RemoveLanguage);

        var plc = 1;
        if(plc > 0) {
            // Add country languages to preferred section
            $(langDropdownSelector.find("li")[plc]).addClass("last_dest");
        }
    });

    function AddLanguage() {
        var language = langSelector.val();
        var level = lvlSelector.val();

        if (language && level && !IsDuplicated(language)) {
            $(".list_items").append(GenerateLanguageElement(language, level));
            $('.icon.close').off("click");
            $('.icon.close').on("click", RemoveLanguage);
            AddSerializedLanguage(language, level);
        }
        ClearValues();
    }

    function RemoveLanguage() {
        var ids = $(this).parent().attr("id").split("|");
        RemoveSerializedLanguage(ids[0], ids[1]);
        ClearValues(defaultLanguage, defaultLevel);
        $(this).parent().remove();
    }

    function GenerateLanguageElement(language, level) {
        var selectedLangText = langDropdownSelector.find("li[val='" + language + "']").html();
        var selectedLvlText = lvlDropdownSelector.find("li[val='" + level + "']").html();

        var template = "<li id='{0}'>{1} - {2}<span class='icon close'></span></li>";
        var result = template.replace("{0}", language + "|" + level);
        result = result.replace("{1}", selectedLangText);
        result = result.replace("{2}", selectedLvlText);

        return result;
    }

    function AddSerializedLanguage(language, level) {
        var selectedLanguages = JSON.parse(addedLanguagesSelector.val());
        selectedLanguages.push({ "Language": language, "Level": level });
        addedLanguagesSelector.val(JSON.stringify(selectedLanguages));
    }

    function RemoveSerializedLanguage(language, level) {
        var selectedLanguages = JSON.parse(addedLanguagesSelector.val());
        selectedLanguages.removeValue("Language", language);
        addedLanguagesSelector.val(JSON.stringify(selectedLanguages));
    }

    function IsDuplicated(language) {
        var selectedLanguages = JSON.parse(addedLanguagesSelector.val());
        for (var i = 0; i < selectedLanguages.length; i++) {
            if (selectedLanguages[i].Language == language) {
                return true;
            }
        }
        return false;
    }

    function ClearValues() {
        langDropdownSelector.children(".s_f").html(defaultLanguage);
        lvlDropdownSelector.children(".s_f").html(defaultLevel);
        langSelector.val('');
        lvlSelector.val('');
    }
